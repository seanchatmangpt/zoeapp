export interface Place {
  id: string;
  label: string;
}
 
export interface Transition {
  id: string;
  label: string;
}
 
export interface Arc {
  id: string;
  source: string; // Place ID or Transition ID
  target: string; // Place ID or Transition ID
  weight: number;
}
 
export type Marking = Record<string, number>; // Place ID -> Token Count
 
export interface PetriNet {
  places: Place[];
  transitions: Transition[];
  arcs: Arc[];
}
 
/**
 * OCEL 2.0 (Object-Centric Event Log) compliant interfaces
 */
export interface OCEL2Event {
  "ocel:id": string;
  "ocel:activity": string;
  "ocel:timestamp": string; // ISO-8601 string
  "ocel:omap": string[]; // List of related object IDs
  "ocel:vmap": Record<string, any>; // Event attributes (must include absolute markdown links for relevant references)
}
 
export interface OCEL2Object {
  "ocel:id": string;
  "ocel:type": string;
  "ocel:vmap": Record<string, any>; // Object attributes
}
 
export interface OCEL2GlobalLog {
  "ocel:attribute-names": string[];
  "ocel:object-types": string[];
  "ocel:event-types": string[];
}
 
export interface OCEL2Log {
  "ocel:global-log": OCEL2GlobalLog;
  "ocel:events": OCEL2Event[];
  "ocel:objects": OCEL2Object[];
}
 
/**
 * Conformance replay results
 */
export interface ReplayTraceResult {
  traceId: string;
  fitness: number;
  consumed: number;
  produced: number;
  missing: number;
  remaining: number;
  missingTokensDetail: Record<string, number>; // Place ID -> missing count
  remainingTokensDetail: Record<string, number>; // Place ID -> remaining count
  violations: SafetyViolation[];
  finalMarking: Marking;
}
 
export interface SafetyViolation {
  type: "CAPACITY_VIOLATION" | "BYPASS_VIOLATION" | "UNAUTHORIZED_TRANSITION" | "FORBIDDEN_CONCURRENT_MARKING";
  description: string;
  timestamp: string;
  details?: Record<string, any>;
}
 
export interface SafetyConstraint {
  maxPlaceCapacity?: Record<string, number>; // Place ID -> max allowed tokens
  forbiddenConcurrent?: string[][]; // List of place IDs that cannot simultaneously have tokens
  requiredPredecessors?: Record<string, string>; // Transition ID -> Place ID that must have had tokens (verifying critical paths)
}
